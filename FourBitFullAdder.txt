# FourBitFullAdder.txt
PARAMS: A0:state A1:state A2:state A3:state \
        B0:state B1:state B2:state B3:state \
        Cin:state Sum0:int Sum1:int Sum2:int Sum3:int C4:int

MAIN-PROCESS FourBitFullAdder
DECLARECHILD TwoBitFullAdder

# initialize carry‐tokens at cycle 0
CREATETOKEN -I C2 C4
SET C2:0 0p
SET C4:0 0p

# bits 0–1 at cycle 0
SET A0:0 $A0
SET A1:0 $A1
SET B0:0 $B0
SET B1:0 $B1
SET Cin:0 $Cin
RUNCHILD TwoBitFullAdder \
  -I $A0 $A1 $B0 $B1 $Cin \
  -O Sum0 Sum1 C2:0
ACCEPTVALS Sum0 Sum1 C2

# advance two cycles → now cycle 2

# bits 2–3 at cycle 2
SET A2:0 $A2
SET A3:0 $A3
SET B2:0 $B2
SET B3:0 $B3
RUNCHILD TwoBitFullAdder \
  -I $A2 $A3 $B2 $B3 C2:2 \
  -O Sum2 Sum3 C4:2
ACCEPTVALS Sum2 Sum3 C4

# cleanup and return all four sums + final carry
DELETETOKEN -I C2 C4
RETURNVALS Sum0 Sum1 Sum2 Sum3 C4
